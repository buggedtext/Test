		async function fetchGH(endpoint, options = {}) {
			const headers = new Headers({
				'Accept': 'application/vnd.github.v3+json',
				'Content-Type': 'application/json'
			});

			if(state.token) {
				headers.set('Authorization', `token ${state.token}`);
			}

			if(options.headers) {
				Object.entries(options.headers).forEach(([key, value]) => {
					headers.set(key, value);
				});
			}

			try {
				const response = await fetch(`${GITHUB_API}${endpoint}`, {
					...options,
					headers
				});

				if(!response.ok) {
					let errorMessage = `HTTP Error: ${response.status}`;

					if(response.status === 404) {
						errorMessage = 'Resource not found';
					} else if(response.status === 403) {
						errorMessage = 'Rate limited or insufficient permissions';
					} else if(response.status === 401) {
						errorMessage = 'Invalid authentication token';
					} else if(response.status === 422) {
						try {
							const errorData = await response.json();
							if(errorData.message) {
								errorMessage = `GitHub API: ${errorData.message}`;
							}
							if(errorData.errors && errorData.errors.length > 0) {
								errorMessage = `GitHub API: ${errorData.errors[0].message}`;
							}
						} catch(e) {
							errorMessage = 'Validation error';
						}
					}

					throw new Error(errorMessage);
				}

				try {
					return await response.json();
				} catch(e) {
					return null;
				}
			} catch(err) {
				console.error('API Error:', err);
				throw err;
			}
		}

		function sortRepositories(repos, sortBy) {
			const sorted = [...repos];
			switch(sortBy) {
				case 'name':
					return sorted.sort((a, b) => a.name.localeCompare(b.name));
				case 'updated':
					return sorted.sort((a, b) => new Date(b.updated_at) - new Date(a.updated_at));
				case 'stars':
					return sorted.sort((a, b) => b.stargazers_count - a.stargazers_count);
				case 'forks':
					return sorted.sort((a, b) => b.forks_count - a.forks_count);
				case 'size':
					return sorted.sort((a, b) => b.size - a.size);
				default:
					return sorted;
			}
		}

		async function loadFiles(path) {
			setLoadingState(true);
			try {
				const encodedPath = path ? encodeURIComponent(path) : '';
				const endpoint = `/repos/${state.currentUser}/${state.currentRepo}/contents/${encodedPath}`;
				const contents = await fetchGH(endpoint);
				
				state.files = Array.isArray(contents) ? contents : (contents ? [contents] : []);
				state.filteredFiles = [...state.files];
				state.currentPath = path;
				sidebarSearchTerm = '';
				
				const sidebarSearch = document.getElementById('sidebar-search');
				if(sidebarSearch) sidebarSearch.value = '';
				
				renderFileList();
				updateBreadcrumbs(path);
				updateFileStats();
			} catch(err) {
				console.error('Failed to load directory:', err);
				state.files = [];
				state.filteredFiles = [];
				renderFileList();
			} finally {
				setLoadingState(false);
			}
		}

		function renderFileList() {
			const container = document.getElementById('sidebar-content');
			if(!container) return;

			const files = sidebarSearchTerm ? state.filteredFiles : state.files;
			let html = '<div class="flex flex-col gap-1">';

			if(state.isOwner && state.token && state.currentRepo) {
				html += `
										<div class="file-creation-card" onclick="showCreateFileModal()">
						<i class="fas fa-file-circle-plus text-xl text-blue-400 mb-1"></i>
						<div class="text-xs font-bold">Create File</div>
					</div>
					<div class="file-creation-card" onclick="showCreateFolderModal()">
						<i class="fas fa-folder-plus text-xl text-yellow-400 mb-1"></i>
						<div class="text-xs font-bold">Create Folder</div>
					</div>
				`;
			}

			if(state.currentPath !== '') {
				html += `
					<div onclick="navigateUp()" class="flex items-center gap-2 p-2 rounded cursor-pointer text-xs text-blue-400 file-item transition-all" style="background-color: rgba(31, 111, 235, 0.05);">
						<i class="fas fa-level-up-alt"></i> .. /
					</div>
				`;
			}

			if(files.length === 0 && state.currentPath === '') {

				html += '<div class="text-center text-gray-600 p-4 mt-10"><i class="fas fa-inbox text-3xl opacity-20 mb-2"></i><p class="text-sm">Empty Repository</p></div>';				html += '</div>';
				container.innerHTML = html;
				return;
			}

			const sorted = files.sort((a, b) => 
				(b.type === 'dir') - (a.type === 'dir') || a.name.localeCompare(b.name)
			);

			html += sorted.map(file => {
				const icon = file.type === 'dir' ? 'fa-folder text-yellow-500' : getFileIcon(file.name);
				const fileCount = file.type === 'dir' ? `<span class="text-[8px]" style="color: var(--text-secondary); margin-left: 4px;">(?)</span>` : '';

				return `
					<div class="flex items-center justify-between p-2 rounded group file-item" style="background-color: rgba(31, 111, 235, 0); border: none;" data-path="${escapeHtml(file.path)}" data-type="${file.type}" data-sha="${file.sha}">
						<div onclick="${file.type === 'dir' ? `loadFiles('${file.path}')` : `openFile('${btoa(file.path)}', '${btoa(file.sha)}')`}" class="flex items-center gap-2 overflow-hidden flex-1 cursor-pointer">
							<i class="fas ${icon} w-4 text-center flex-shrink-0"></i>
							<span class="text-xs truncate">${escapeHtml(file.name)}</span>
							${fileCount}
						</div>
						<div class="flex gap-2 opacity-0 group-hover:opacity-100 transition">
							${file.type === 'file' && state.isOwner ? `<i class="fas fa-pen-to-square text-[10px] text-blue-400 hover:scale-110 cursor-pointer action-button" onclick="showEditFileNameModal('${escapeHtml(file.path)}'); event.stopPropagation();"></i>` : ''}
							${state.isOwner ? `<i class="fas fa-trash-alt text-[10px] text-red-500 hover:scale-110 cursor-pointer action-button" onclick="refreshAndDelete('${escapeHtml(file.path)}', '${file.sha}', event)"></i>` : ''}
							<i class="fas fa-external-link-square-alt text-[10px] text-gray-500 hover:text-white cursor-pointer action-button" onclick="window.open('${escapeHtml(file.html_url)}', '_blank'); event.stopPropagation();"></i>
						</div>
					</div>
				`;
			}).join('');

			html += '</div>';
			container.innerHTML = html;
		}

		async function openFile(encodedPath, encodedSha) {
			const path = atob(encodedPath);
			const sha = atob(encodedSha);
			state.currentPath = path;
			state.currentSHA = sha;

			const file = state.files.find(f => f.path === path);
			if(!file) return;

			const label = document.getElementById('active-file-label');
			if(label) {
				label.innerHTML = `<i class="fas ${getFileIcon(path)}"></i> ${escapeHtml(path.split('/').pop())}`;
			}

			const isImage = /\.(jpg|jpeg|png|gif|webp|svg)$/i.test(path);

			if(isImage) {
				showImage(file.download_url);
				const copyBtn = document.getElementById('copy-code-btn');
				if(copyBtn) copyBtn.classList.add('hidden');
				const previewBtn = document.getElementById('preview-btn');
				if(previewBtn) previewBtn.classList.add('hidden');
			} else {
				await showEditor(file.download_url, path);
				const copyBtn = document.getElementById('copy-code-btn');
				if(copyBtn) copyBtn.classList.remove('hidden');
				const previewBtn = document.getElementById('preview-btn');
				if(previewBtn && /\.(html|md)$/i.test(path)) previewBtn.classList.remove('hidden');
			}

			toggleEditButtons();
		}

		async function showEditor(url, path) {
			document.getElementById('image-viewport').classList.add('hidden');
			document.getElementById('preview-viewport').classList.add('hidden');
			document.getElementById('editor-viewport').parentElement.classList.remove('hidden');

			try {
				const response = await fetch(url);
				
				if(!response.ok) {
					throw new Error(`HTTP ${response.status}`);
				}

				const size = parseInt(response.headers.get('content-length') || '0');
				const MAX_FILE_SIZE = 5 * 1024 * 1024;

				if(size > MAX_FILE_SIZE) {
					state.monacoEditor.setValue(`File too large (${(size / 1024 / 1024).toFixed(2)}MB > 5MB limit)`);
					state.monacoEditor.updateOptions({ readOnly: true });
					return;
				}

				const content = await response.text();
				state.monacoEditor.setValue(content);

				const ext = path.split('.').pop().toLowerCase();
				const lang = getLanguageByExt(ext);
				monaco.editor.setModelLanguage(state.monacoEditor.getModel(), lang);

				state.monacoEditor.updateOptions({
					readOnly: !state.isOwner
				});
			} catch(err) {
				console.error('Error loading file:', err);
				state.monacoEditor.setValue(`Error loading file: ${err.message}`);
				state.monacoEditor.updateOptions({ readOnly: true });
			}
		}

		function showImage(url) {
			document.getElementById('editor-viewport').parentElement.classList.add('hidden');
			document.getElementById('preview-viewport').classList.add('hidden');
			const imgViewport = document.getElementById('image-viewport');
			imgViewport.classList.remove('hidden');
			
			const previewImg = document.getElementById('preview-img');
			if(previewImg) previewImg.src = url;
			
			const downloadLink = document.getElementById('img-download-link');
			if(downloadLink) {
				downloadLink.href = url;
				downloadLink.download = state.currentPath.split('/').pop();
			}
		}

		async function saveFileContent() {
			if(!state.isOwner || !state.currentPath) {
				alert('No file selected or insufficient permissions');
				return;
			}

			const newContent = state.monacoEditor.getValue();
			const encodedContent = btoa(unescape(encodeURIComponent(newContent)));

			setLoadingState(true);
			try {
				const result = await fetchGH(
					`/repos/${state.currentUser}/${state.currentRepo}/contents/${state.currentPath}`,
					{
						method: 'PUT',
						body: JSON.stringify({
							message: `Update ${state.currentPath.split('/').pop()}`,
							content: encodedContent,
							sha: state.currentSHA
						})
					}
				);

				state.currentSHA = result.content.sha;
				showNotification(`<i class="fas fa-check-circle"></i> File saved!`);
				addNotification(`File updated: ${state.currentPath}`);
			} catch(err) {
				alert(`Save failed: ${err.message}`);
			} finally {
				setLoadingState(false);
			}
		}

		async function deleteFileWithValidation(path, expectedSha, event) {
			if(event) {
				event.stopPropagation();
			}
			
			const fileName = path.split('/').pop();
			if(!confirm(`Delete "${fileName}"?`)) return;

			if(!state.token) {
				alert('PAT token required');
				return;
			}

			setLoadingState(true);
			try {
				let fileData;
				try {
					fileData = await fetchGH(
						`/repos/${state.currentUser}/${state.currentRepo}/contents/${encodeURIComponent(path)}`
					);
				} catch(err) {
					if(err.message.includes('404')) {
						throw new Error('File not found');
					}
					throw err;
				}

				if(expectedSha && fileData.sha !== expectedSha) {
					const confirmOverride = confirm(
						'File has been modified. Delete anyway?'
					);
					if(!confirmOverride) {
						throw new Error('Delete cancelled');
					}
				}

				await fetchGH(
					`/repos/${state.currentUser}/${state.currentRepo}/contents/${encodeURIComponent(path)}`,
					{
						method: 'DELETE',
						body: JSON.stringify({
							message: `Delete ${fileName}`,
							sha: fileData.sha,
							committer: {
								name: 'GitNexus',
								email: 'gitnexus@app.local'
							}
						})
					}
				);

				await loadFiles(state.currentPath);
				showNotification(`<i class="fas fa-trash"></i> "${fileName}" deleted!`);
				addNotification(`File deleted: ${path}`);
			} catch(err) {
				console.error('Delete error:', err);
				showNotification(`<i class="fas fa-exclamation-circle"></i> ${err.message}`);
			} finally {
				setLoadingState(false);
			}
		}

		async function refreshAndDelete(path, sha, event) {
			event.stopPropagation();
			
			setLoadingState(true);
			try {
				await loadFiles(state.currentPath);
				
				const updatedFile = state.files.find(f => f.path === path);
				if(!updatedFile) {
					throw new Error('File not found');
				}

				await deleteFileWithValidation(path, updatedFile.sha);
			} catch(err) {
				showNotification(`<i class="fas fa-exclamation-circle"></i> ${err.message}`);
			} finally {
				setLoadingState(false);
			}
		}

			if(commitMsgInput) commitMsgInput.value = `Rename ${currentName}`;

		async function confirmRenameFile() {
			const modal = window.edit_filename_modal;
			const originalPath = modal.dataset.originalPath;
			const newFilename = document.getElementById('new-filename').value.trim();
			const commitMsg = document.getElementById('rename-commit-msg').value.trim() || 'Rename file';

			if(!newFilename) {
				alert('Please enter a new filename');
				return;
			}

			if(newFilename === originalPath.split('/').pop()) {
				alert('New filename must be different');
				return;
			}

			setLoadingState(true);
			try {
				const fileData = await fetchGH(
					`/repos/${state.currentUser}/${state.currentRepo}/contents/${encodeURIComponent(originalPath)}`
				);

				const fileContent = await (await fetch(fileData.download_url)).text();
				const pathParts = originalPath.split('/');
				pathParts.pop();
				const directory = pathParts.join('/');
				const newPath = directory ? `${directory}/${newFilename}` : newFilename;

				const encodedContent = btoa(unescape(encodeURIComponent(fileContent)));
				await fetchGH(`/repos/${state.currentUser}/${state.currentRepo}/contents/${newPath}`, {
					method: 'PUT',
					body: JSON.stringify({
						message: commitMsg,
						content: encodedContent
					})
				});

				await fetchGH(
					`/repos/${state.currentUser}/${state.currentRepo}/contents/${encodeURIComponent(originalPath)}`,
					{
						method: 'DELETE',
						body: JSON.stringify({
							message: commitMsg,
							sha: fileData.sha
						})
					}
				);

				modal.close();
				showNotification(`<i class="fas fa-check-circle"></i> File renamed!`);
				addNotification(`File renamed: ${originalPath} → ${newPath}`);
				await loadFiles(state.currentPath);
			} catch(err) {
				alert(`Failed: ${err.message}`);
			} finally {
				setLoadingState(false);
			}
		}

					fullPath = filePath.endsWith('/') ? `${filePath}${fileName}` : `${filePath}/${fileName}`;

		async function performUpload() {
			const fileInput = document.getElementById('file-picker');
			const files = fileInput.files;

			if(files.length === 0) {
				alert('Please select at least one file');
				return;
			}

			const uploadBtn = document.getElementById('upload-btn');
			if(uploadBtn) uploadBtn.disabled = true;

			document.getElementById('upload-progress').classList.remove('hidden');
			const uploadTotal = document.getElementById('upload-total');
			if(uploadTotal) uploadTotal.innerText = files.length;

			let uploadedCount = 0;
			const MAX_FILE_SIZE = 100 * 1024 * 1024;

			for(const file of files) {
				try {
					if(file.size > MAX_FILE_SIZE) {
						showNotification(`<i class="fas fa-exclamation-triangle"></i> File ${file.name} exceeds limit`);
						continue;
					}

					await uploadFile(file);
					uploadedCount++;
					const uploadCurrent = document.getElementById('upload-current');
					if(uploadCurrent) uploadCurrent.innerText = uploadedCount;
				} catch(err) {
					console.error(`Upload failed for ${file.name}:`, err);
					showNotification(`<i class="fas fa-exclamation-circle"></i> Failed to upload ${file.name}`);
				}
			}

			if(window.upload_modal) window.upload_modal.close();
			fileInput.value = '';
			if(uploadBtn) uploadBtn.disabled = false;
			await loadFiles(state.currentPath);
			showNotification(`<i class="fas fa-check-circle"></i> ${uploadedCount}/${files.length} files uploaded!`);
			addNotification(`${uploadedCount} files uploaded to ${state.currentRepo}`);
		}

		async function uploadFile(file) {
			return new Promise((resolve, reject) => {
				const reader = new FileReader();

				reader.onload = async(e) => {
					try {
						const content = e.target.result.split(',')[1];
						const uploadPath = state.currentPath ? `${state.currentPath}/${file.name}` : file.name;

						await fetchGH(`/repos/${state.currentUser}/${state.currentRepo}/contents/${uploadPath}`, {
							method: 'PUT',
							body: JSON.stringify({
								message: `Upload ${file.name}`,
								content: content
							})
						});

						resolve();
					} catch(err) {
						reject(err);
					}
				};

				reader.onerror = () => reject(new Error('File read error'));
				reader.readAsDataURL(file);
			});
		}

		function showMarkdownPreview() {
			const markdown = state.monacoEditor.getValue();
			const previewViewport = document.getElementById('preview-viewport');
			
			let html = markdown
				.replace(/^### (.*?)$/gm, '<h3>$1</h3>')
				.replace(/^## (.*?)$/gm, '<h2>$1</h2>')
				.replace(/^# (.*?)$/gm, '<h1>$1</h1>')
				.replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
				.replace(/\*(.*?)\*/g, '<em>$1</em>')
				.replace(/\n/g, '<br>');

			previewViewport.innerHTML = `<div style="padding: 20px; font-size: 14px; line-height: 1.6;">${html}</div>`;
			previewViewport.classList.remove('hidden');
			document.getElementById('editor-viewport').parentElement.classList.add('hidden');
		}

		function updateBreadcrumbs(path) {
			const container = document.getElementById('breadcrumb-container');
			if(!container) return;

			const parts = path.split('/').filter(p => p);

			let html = `
				<span onclick="loadFiles('')" class="hover:text-blue-400 cursor-pointer code-link">
					<i class="fas fa-folder-open mr-1"></i>${escapeHtml(state.currentRepo || 'Repository')}
				</span>
			`;

			let currentPathBuild = '';
			parts.forEach((part, index) => {

				html += `

					<span onclick="loadFiles('${currentPathBuild}')" class="hover:text-blue-400 cursor-pointer code-link">${escapeHtml(part)}</span>
				`;
			});

			container.innerHTML = html;
		}

			clearTimeout(searchDebounceTimer);
			searchDebounceTimer = setTimeout(() => {

				sidebarSearchTerm = searchTerm;

				if(state.activeTab === 'repos') {
					state.filteredRepos = state.repos.filter(repo => 
						repo.name.toLowerCase().includes(searchTerm) || 
						(repo.description && repo.description.toLowerCase().includes(searchTerm))
					);
				} else if(state.activeTab === 'files') {
					state.filteredFiles = state.files.filter(file => 
						file.name.toLowerCase().includes(searchTerm)
					);
				}

				if(state.activeTab === 'repos') renderRepoList();
				else if(state.activeTab === 'files') renderFileList();

		async function askNexusBot() {
			const prompt = document.getElementById('copilot-prompt').value.trim();
			if(!prompt) {
				alert('Please enter a question');
				return;
			}

			if(!state.copilotConnected) {
				alert('Please connect NexusBot first');
				return;
			}

			setLoadingState(true);
			try {
				const responses = {
					'explain': 'This code defines a function that processes user input and returns formatted output.',
					'optimize': 'Consider using a Set instead of an array for O(1) lookups, and memoize repeated calculations.',
					'fix': 'The bug is likely in line 42 where the array bounds are not checked before access.',
					'document': '/**\n * Processes input data and returns formatted result\n * @param {string} data - Input data to process\n * @returns {Object} Formatted result object\n */\n'
				};

				const response = responses[Object.keys(responses).find(key => prompt.toLowerCase().includes(key))] || 
					'NexusBot analysis: The code appears to be well-structured. Consider adding more error handling and input validation.';

				document.getElementById('copilot-response').classList.remove('hidden');
				document.getElementById('copilot-output').innerText = response;
				document.getElementById('copilot-prompt').value = '';
				showNotification(`<i class="fas fa-lightbulb"></i> NexusBot suggestion received!`);
			} catch(err) {
				alert(`Failed: ${err.message}`);
			} finally {
				setLoadingState(false);
			}
		}

		function quickCommand(command) {

				'document': 'Add documentation to this code'

		function searchMusic() {

			if(!searchTerm) return;

				{ title: 'Focus Mode', artist: 'Lo-Fi Hip Hop', duration: 'infinite' },
				{ title: 'Coding Vibes', artist: 'Electronic', duration: '2:45' },

		function downloadFile() {
			if(!state.currentPath) {
				alert('Please select a file first');
				return;
			}

			const isImage = /\.(jpg|jpeg|png|gif|webp|svg)$/i.test(state.currentPath);

			if(isImage) {
				const downloadLink = document.getElementById('img-download-link');
				if(downloadLink) downloadLink.click();
			} else {
				const filename = state.currentPath.split('/').pop();
				const content = state.monacoEditor.getValue();
				const blob = new Blob([content], { type: 'text/plain' });
				const url = window.URL.createObjectURL(blob);
				const a = document.createElement('a');
				a.href = url;
				a.download = filename;
				a.click();
				window.URL.revokeObjectURL(url);
			}

			showNotification(`<i class="fas fa-download"></i> Download started!`);
			addNotification(`Downloaded: ${state.currentPath}`);
		}

		function copyRawUrl() {

function getFileIcon(filename) {
    const ext = filename.split('.').pop().toLowerCase();
    const icons = {
        js: 'fa-brands fa-js text-yellow-400',
        ts: 'fa-brands fa-square-js text-blue-400',
        html: 'fa-brands fa-html5 text-orange-500',
        css: 'fa-brands fa-css3-alt text-blue-500',
        json: 'fa-solid fa-file-code text-yellow-600',
        md: 'fa-brands fa-markdown text-gray-400',
        py: 'fa-brands fa-python text-blue-300',
        rb: 'fa-solid fa-gem text-red-500',
        java: 'fa-brands fa-java text-orange-600',
        cpp: 'fa-solid fa-plus text-blue-600',
        sh: 'fa-solid fa-terminal text-green-500',
        yml: 'fa-solid fa-file-code text-red-400',
        yaml: 'fa-solid fa-file-code text-red-400',
        default: 'fa-solid fa-file-alt text-gray-400'
    };
    return icons[ext] || icons.default;
}
