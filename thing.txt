		// API Helper
		async function fetchGH(endpoint, options = {}) {
			const headers = new Headers({
				'Accept': 'application/vnd.github.v3+json',
				'Content-Type': 'application/json'
			});

			if(state.token) {
				headers.set('Authorization', `token ${state.token}`);
			}

			if(options.headers) {
				Object.entries(options.headers).forEach(([key, value]) => {
					headers.set(key, value);
				});
			}

			try {
				const response = await fetch(`${GITHUB_API}${endpoint}`, {
					...options,
					headers
				});

				if(!response.ok) {
					let errorMessage = `HTTP ${response.status}`;
					if(response.status === 404) errorMessage = 'Resource not found';
					else if(response.status === 403) errorMessage = 'Rate limited or permission denied';
					else if(response.status === 401) errorMessage = 'Invalid authentication token';

					throw new Error(errorMessage);
				}

				const contentType = response.headers.get('content-type');
				if(!contentType || !contentType.includes('application/json')) {
					return null;
				}
				
				return await response.json();
			} catch(err) {
				console.error('API Error:', err);
				throw err;
			}
		}

		function sortRepositories(repos, sortBy) {
			const sorted = [...repos];
			switch(sortBy) {
				case 'name':
					return sorted.sort((a, b) => a.name.localeCompare(b.name));
				case 'updated':
					return sorted.sort((a, b) => new Date(b.updated_at) - new Date(a.updated_at));
				case 'stars':
					return sorted.sort((a, b) => b.stargazers_count - a.stargazers_count);
				case 'forks':
					return sorted.sort((a, b) => b.forks_count - a.forks_count);
				case 'size':
					return sorted.sort((a, b) => b.size - a.size);
				default:
					return sorted;
			}
		}

		async function saveDescriptionChanges() {
			const modal = window.desc_editor_modal;
			const repoName = modal.dataset.repoName;
			const textarea = document.getElementById('desc-textarea');
			const patInput = document.getElementById('desc-pat-input');
			
			const newDescription = textarea.value.trim();
			const pat = patInput.value;

			if(!pat) {
				alert('Please enter your PAT token');
				return;
			}

			if(newDescription.length > 300) {
				alert('Description must be 300 characters or less');
				return;
			}

			setLoadingState(true);
			try {
				const response = await fetch(`${GITHUB_API}/repos/${state.currentUser}/${repoName}`, {
					method: 'PATCH',
					headers: {
						'Authorization': `token ${pat}`,
						'Accept': 'application/vnd.github.v3+json',
						'Content-Type': 'application/json'
					},
					body: JSON.stringify({ description: newDescription })
				});

				if(!response.ok) {
					if(response.status === 401) throw new Error('Invalid PAT token');
					if(response.status === 403) throw new Error('Permission denied');
					if(response.status === 404) throw new Error('Repository not found');
					throw new Error(`HTTP ${response.status}`);
				}

				const repoIndex = state.repos.findIndex(r => r.name === repoName);
				if(repoIndex !== -1) {
					state.repos[repoIndex].description = newDescription;
					state.filteredRepos = [...state.repos];
				}

				const contentDisplay = document.getElementById('desc-content-display');
				if(contentDisplay) contentDisplay.innerText = newDescription || 'No description provided.';

				switchToDescViewMode();
				renderRepoList();
				showNotification(`<i class="fas fa-check-circle"></i> Description updated!`);
				addNotification(`Updated description for "${repoName}"`);
			} catch(err) {
				alert(`Failed: ${err.message}`);
			} finally {
				setLoadingState(false);
			}
		}

		// File Management
		async function loadFiles(path) {
			setLoadingState(true);
			try {
				const encodedPath = path ? encodeURIComponent(path) : '';
				const endpoint = `/repos/${state.currentUser}/${state.currentRepo}/contents/${encodedPath}`;
				const contents = await fetchGH(endpoint);
				
				state.files = Array.isArray(contents) ? contents : (contents ? [contents] : []);
				state.filteredFiles = [...state.files];
				state.currentPath = path;
				sidebarSearchTerm = '';
				
				const sidebarSearch = document.getElementById('sidebar-search');
				if(sidebarSearch) sidebarSearch.value = '';
				
				renderFileList();
				updateBreadcrumbs(path);
				updateFileStats();
			} catch(err) {
				console.error('Failed to load directory:', err);
				state.files = [];
				state.filteredFiles = [];
				renderFileList();
			} finally {
				setLoadingState(false);
			}
		}

		function renderFileList() {
			const container = document.getElementById('sidebar-content');
			if(!container) return;

			const files = sidebarSearchTerm ? state.filteredFiles : state.files;
			let html = '<div class="flex flex-col gap-1">';

			if(state.isOwner && state.token && state.currentRepo) {
				html += `
					<div class="p-2 rounded-lg border-2 border-dashed text-center cursor-pointer transition hover:shadow-lg hover:shadow-blue-500/20 mb-2" style="border-color: var(--accent); background-color: rgba(31, 111, 235, 0.05);" onclick="showCreateFileModal()">
						<i class="fas fa-file-plus text-lg text-blue-400 mb-1"></i>
						<div class="text-xs font-bold">New File</div>
					</div>
					<div class="p-2 rounded-lg border-2 border-dashed text-center cursor-pointer transition hover:shadow-lg hover:shadow-yellow-500/20 mb-2" style="border-color: var(--accent); background-color: rgba(31, 111, 235, 0.05);" onclick="showCreateFolderModal()">
						<i class="fas fa-folder-plus text-lg text-yellow-400 mb-1"></i>
						<div class="text-xs font-bold">New Folder</div>
					</div>
				`;
			}

			if(state.currentPath !== '') {
				html += `
					<div onclick="navigateUp()" class="flex items-center gap-2 p-2 rounded cursor-pointer text-xs file-item transition" style="background-color: rgba(31, 111, 235, 0.05);">
						<i class="fas fa-level-up-alt"></i> <span class="text-[var(--accent)]">.. /</span>
					</div>
				`;
			}

			if(files.length === 0 && state.currentPath === '') {
				html += '<div class="text-center text-gray-600 p-4 mt-10"><i class="fas fa-inbox text-3xl opacity-20 mb-2"></i><p class="text-sm">Empty</p></div>';
				html += '</div>';
				container.innerHTML = html;
				return;
			}

			const sorted = files.sort((a, b) => 
				(b.type === 'dir') - (a.type === 'dir') || a.name.localeCompare(b.name)
			);

			html += sorted.map(file => {
				const icon = file.type === 'dir' ? 'fa-folder text-yellow-400' : getFileIcon(file.name);

				return `
					<div class="flex items-center justify-between p-2 rounded group file-item transition" style="background-color: transparent;">
						<div onclick="${file.type === 'dir' ? `loadFiles('${file.path}')` : `openFile('${btoa(file.path)}', '${btoa(file.sha)}')`}" class="flex items-center gap-2 overflow-hidden flex-1 cursor-pointer">

						</div>
						<div class="flex gap-2 opacity-0 group-hover:opacity-100 transition">

							<i class="fas fa-arrow-up-right-from-square text-[10px] cursor-pointer hover:text-[var(--accent)]" onclick="window.open('${escapeHtml(file.html_url)}', '_blank'); event.stopPropagation();"></i>
						</div>
					</div>
				`;
			}).join('');

			html += '</div>';
			container.innerHTML = html;
		}

		async function openFile(encodedPath, encodedSha) {
			const path = atob(encodedPath);
			const sha = atob(encodedSha);
			state.currentPath = path;
			state.currentSHA = sha;

			const file = state.files.find(f => f.path === path);
			if(!file) return;

			const label = document.getElementById('active-file-label');
			if(label) {
				label.innerHTML = `<i class="fas ${getFileIcon(path)}"></i> ${escapeHtml(path.split('/').pop())}`;
			}

			const isImage = /\.(jpg|jpeg|png|gif|webp|svg)$/i.test(path);

			if(isImage) {
				showImage(file.download_url);
				const copyBtn = document.getElementById('copy-code-btn');
				if(copyBtn) copyBtn.classList.add('hidden');
				const previewBtn = document.getElementById('preview-btn');
				if(previewBtn) previewBtn.classList.add('hidden');
			} else {
				await showEditor(file.download_url, path);
				const copyBtn = document.getElementById('copy-code-btn');
				if(copyBtn) copyBtn.classList.remove('hidden');
				const previewBtn = document.getElementById('preview-btn');
				if(previewBtn && /\.(html|md|markdown)$/i.test(path)) previewBtn.classList.remove('hidden');
			}

			toggleEditButtons();
		}

		async function showEditor(url, path) {
			document.getElementById('image-viewport').classList.add('hidden');
			document.getElementById('preview-viewport').classList.add('hidden');
			document.getElementById('editor-viewport').parentElement.style.display = 'block';

			try {
				const response = await fetch(url);
				
				if(!response.ok) {
					throw new Error(`HTTP ${response.status}`);
				}

				const size = parseInt(response.headers.get('content-length') || '0');
				const MAX_FILE_SIZE = 5 * 1024 * 1024;

				if(size > MAX_FILE_SIZE) {
					state.monacoEditor.setValue(`File too large to edit\n${(size / 1024 / 1024).toFixed(2)}MB > 5MB limit`);
					state.monacoEditor.updateOptions({ readOnly: true });
					return;
				}

				const content = await response.text();
				state.monacoEditor.setValue(content);

				const ext = path.split('.').pop().toLowerCase();
				const lang = getLanguageByExt(ext);
				monaco.editor.setModelLanguage(state.monacoEditor.getModel(), lang);

				state.monacoEditor.updateOptions({
					readOnly: !state.isOwner
				});
			} catch(err) {
				console.error('Error loading file:', err);
				state.monacoEditor.setValue(`Error loading file:\n${err.message}`);
				state.monacoEditor.updateOptions({ readOnly: true });
			}
		}

		function showImage(url) {
			document.getElementById('editor-viewport').parentElement.style.display = 'none';
			document.getElementById('preview-viewport').classList.add('hidden');
			const imgViewport = document.getElementById('image-viewport');
			imgViewport.classList.remove('hidden');
			
			const previewImg = document.getElementById('preview-img');
			if(previewImg) previewImg.src = url;
			
			const downloadLink = document.getElementById('img-download-link');
			if(downloadLink) {
				downloadLink.href = url;
				downloadLink.download = state.currentPath.split('/').pop();
			}
		}

		async function saveFileContent() {
			if(!state.isOwner || !state.currentPath) {
				alert('No file selected or insufficient permissions');
				return;
			}

			const newContent = state.monacoEditor.getValue();
			const encodedContent = btoa(unescape(encodeURIComponent(newContent)));

			setLoadingState(true);
			try {
				const result = await fetchGH(
					`/repos/${state.currentUser}/${state.currentRepo}/contents/${state.currentPath}`,
					{
						method: 'PUT',
						body: JSON.stringify({
							message: `Update ${state.currentPath.split('/').pop()}`,
							content: encodedContent,
							sha: state.currentSHA
						})
					}
				);

				state.currentSHA = result.content.sha;
				showNotification(`<i class="fas fa-check-circle"></i> File saved!`);
				addNotification(`File updated: ${state.currentPath}`);
			} catch(err) {
				alert(`Save failed: ${err.message}`);
			} finally {
				setLoadingState(false);
			}
		}

		async function deleteFileWithValidation(path, expectedSha, event) {
			if(event) event.stopPropagation();
			
			const fileName = path.split('/').pop();
			if(!confirm(`Delete "${fileName}"?`)) return;

			if(!state.token) {
				alert('PAT token required');
				return;
			}

			setLoadingState(true);
			try {
				let fileData;
				try {
					fileData = await fetchGH(
						`/repos/${state.currentUser}/${state.currentRepo}/contents/${encodeURIComponent(path)}`
					);
				} catch(err) {
					if(err.message.includes('404')) {
						throw new Error('File not found');
					}
					throw err;
				}

				if(expectedSha && fileData.sha !== expectedSha) {
					const confirmOverride = confirm('File modified. Delete anyway?');
					if(!confirmOverride) throw new Error('Delete cancelled');
				}

				await fetchGH(
					`/repos/${state.currentUser}/${state.currentRepo}/contents/${encodeURIComponent(path)}`,
					{
						method: 'DELETE',
						body: JSON.stringify({
							message: `Delete ${fileName}`,
							sha: fileData.sha
						})
					}
				);

				await loadFiles(state.currentPath);
				showNotification(`<i class="fas fa-trash"></i> "${fileName}" deleted!`);
				addNotification(`File deleted: ${path}`);
			} catch(err) {
				console.error('Delete error:', err);
				showNotification(`<i class="fas fa-exclamation-circle"></i> ${err.message}`);
			} finally {
				setLoadingState(false);
			}
		}

		async function refreshAndDelete(path, sha, event) {
			event.stopPropagation();
			
			setLoadingState(true);
			try {
				await loadFiles(state.currentPath);
				
				const updatedFile = state.files.find(f => f.path === path);
				if(!updatedFile) {
					throw new Error('File not found');
				}

				await deleteFileWithValidation(path, updatedFile.sha);
			} catch(err) {
				showNotification(`<i class="fas fa-exclamation-circle"></i> ${err.message}`);
			} finally {
				setLoadingState(false);
			}
		}

		async function confirmRenameFile() {
			const modal = window.edit_filename_modal;
			const originalPath = modal.dataset.originalPath;
			const newFilename = document.getElementById('new-filename').value.trim();
			const commitMsg = document.getElementById('rename-commit-msg').value.trim() || 'Rename file';

			if(!newFilename) {
				alert('Please enter a new filename');
				return;
			}

			if(newFilename === originalPath.split('/').pop()) {
				alert('New filename must be different');
				return;
			}

			setLoadingState(true);
			try {
				const fileData = await fetchGH(
					`/repos/${state.currentUser}/${state.currentRepo}/contents/${encodeURIComponent(originalPath)}`
				);

				const fileContent = await (await fetch(fileData.download_url)).text();
				const pathParts = originalPath.split('/');
				pathParts.pop();
				const directory = pathParts.join('/');
				const newPath = directory ? `${directory}/${newFilename}` : newFilename;

				const encodedContent = btoa(unescape(encodeURIComponent(fileContent)));
				await fetchGH(`/repos/${state.currentUser}/${state.currentRepo}/contents/${newPath}`, {
					method: 'PUT',
					body: JSON.stringify({
						message: commitMsg,
						content: encodedContent
					})
				});

				await fetchGH(
					`/repos/${state.currentUser}/${state.currentRepo}/contents/${encodeURIComponent(originalPath)}`,
					{
						method: 'DELETE',
						body: JSON.stringify({
							message: commitMsg,
							sha: fileData.sha
						})
					}
				);

				modal.close();
				showNotification(`<i class="fas fa-check-circle"></i> File renamed!`);
				addNotification(`File renamed: ${originalPath.split('/').pop()} → ${newFilename}`);
				await loadFiles(state.currentPath);
			} catch(err) {
				alert(`Failed: ${err.message}`);
			} finally {
				setLoadingState(false);
			}
		}

		async function performUpload() {
			const fileInput = document.getElementById('file-picker');
			const files = fileInput.files;

			if(files.length === 0) {
				alert('Please select files');
				return;
			}

			const uploadBtn = document.getElementById('upload-btn');
			if(uploadBtn) uploadBtn.disabled = true;

			document.getElementById('upload-progress').classList.remove('hidden');
			if(document.getElementById('upload-total')) document.getElementById('upload-total').innerText = files.length;

			let uploadedCount = 0;
			const MAX_FILE_SIZE = 100 * 1024 * 1024;
			const details = document.getElementById('upload-details');

			for(const file of files) {
				try {
					if(file.size > MAX_FILE_SIZE) {
						showNotification(`<i class="fas fa-exclamation-triangle"></i> ${file.name} exceeds limit`);
						continue;
					}

					if(details) details.innerHTML += `<div>↳ ${escapeHtml(file.name)}...</div>`;
					await uploadFile(file);
					uploadedCount++;
					if(document.getElementById('upload-current')) document.getElementById('upload-current').innerText = uploadedCount;
					if(details) details.innerHTML += `<div style="color: #22c55e;">✓ ${escapeHtml(file.name)}</div>`;
				} catch(err) {
					console.error(`Upload failed for ${file.name}:`, err);
					if(details) details.innerHTML += `<div style="color: #ef4444;">✗ ${escapeHtml(file.name)}</div>`;
				}
			}

			if(window.upload_modal) window.upload_modal.close();
			fileInput.value = '';
			if(uploadBtn) uploadBtn.disabled = false;
			await loadFiles(state.currentPath);
			showNotification(`<i class="fas fa-check-circle"></i> ${uploadedCount}/${files.length} uploaded!`);
			addNotification(`${uploadedCount} files uploaded`);
		}

		async function uploadFile(file) {
			return new Promise((resolve, reject) => {
				const reader = new FileReader();

				reader.onload = async(e) => {
					try {
						const content = e.target.result.split(',')[1];
						const uploadPath = state.currentPath ? `${state.currentPath}/${file.name}` : file.name;

						await fetchGH(`/repos/${state.currentUser}/${state.currentRepo}/contents/${uploadPath}`, {
							method: 'PUT',
							body: JSON.stringify({
								message: `Upload ${file.name}`,
								content: content
							})
						});

						resolve();
					} catch(err) {
						reject(err);
					}
				};

				reader.onerror = () => reject(new Error('File read error'));
				reader.readAsDataURL(file);
			});
		}

		function updateBreadcrumbs(path) {
			const container = document.getElementById('breadcrumb-container');
			if(!container) return;

			const parts = path.split('/').filter(p => p);

			let html = `
				<span onclick="loadFiles('')" class="hover:text-[var(--accent)] cursor-pointer transition">
					<i class="fas fa-folder-open mr-1"></i>${escapeHtml(state.currentRepo || 'Repository')}
				</span>
			`;

			let currentPathBuild = '';
			parts.forEach((part, index) => {
				currentPathBuild += (index === 0 ? '' : '/') + part;
				html += `
					<i class="fas fa-chevron-right text-[8px]" style="color: var(--text-secondary);"></i>
					<span onclick="loadFiles('${currentPathBuild}')" class="hover:text-[var(--accent)] cursor-pointer transition">${escapeHtml(part)}</span>
				`;
			});

			container.innerHTML = html;
		}

		function filterSidebarContent() {
			const searchTerm = document.getElementById('sidebar-search').value.toLowerCase();
			sidebarSearchTerm = searchTerm;

			if(state.activeTab === 'repos') {
				state.filteredRepos = state.repos.filter(repo => 
					repo.name.toLowerCase().includes(searchTerm) || 
					(repo.description && repo.description.toLowerCase().includes(searchTerm))
				);

		// Utility
		function downloadFile() {
			if(!state.currentPath) {
				alert('Select a file first');
				return;
			}

			const isImage = /\.(jpg|jpeg|png|gif|webp|svg)$/i.test(state.currentPath);

			if(isImage) {
				const downloadLink = document.getElementById('img-download-link');
				if(downloadLink) downloadLink.click();
			} else {
				const filename = state.currentPath.split('/').pop();
				const content = state.monacoEditor.getValue();
				const blob = new Blob([content], { type: 'text/plain' });
				const url = window.URL.createObjectURL(blob);
				const a = document.createElement('a');
				a.href = url;
				a.download = filename;
				a.click();
				window.URL.revokeObjectURL(url);
			}

			showNotification(`<i class="fas fa-download"></i> Download started!`);
			addNotification(`Downloaded: ${state.currentPath}`);
		}

					if(e.key === 'k') {
